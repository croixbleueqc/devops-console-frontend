import { MockComponent } from './authentication-context.composition';

## React Theme Context

Use this context component to manager authentication state of application.

### Component usage

```tsx
import React, { useContext } from 'react';
import { AuthenticationProvider, AuthenticationProviderProps } from './theme-context-provider';
import { MockComponent } from './authentication-context.composition';

const defaultState: AuthenticationProviderProps['defaultState'] = {
    isConnected: false,
};

<AuthenticationProvider isConnected={defaultState}>
    // User is not connected!
    <MockComponent />
</AuthenticationProvider>;
```

### Using props to customize authentication status

Get connected user:

```tsx live
() => {
    return (
        <AuthenticationProvider
            defaultState={{
                isConnected: true,
            }}
        >
            <MockComponent />
        </AuthenticationProvider>
    );
};
```
